extends Node2D

const CELL_SIZE = 64
const CELL_INDENT = 0.9
const BOARD_SIZE = 8
const BOARD_SHIFT_X = 100
const BOARD_SHIFT_Y = 200
const GEAR_TEXTURE_SIZE = 509

var cells: Array = []

func _ready():
	generate_board()

func generate_board():
	for child in get_children():
		child.queue_free()
	
	cells.clear()
	
	for x in range(BOARD_SIZE):
		var row = []
		for y in range(BOARD_SIZE):
			var cell = preload("res://scenes/Cell.tscn").instantiate()
			add_child(cell)
			cell.position = Vector2(x * CELL_SIZE + BOARD_SHIFT_X, (BOARD_SIZE - 1 - y) * CELL_SIZE + BOARD_SHIFT_Y)
			cell.board_pos = Vector2i(x, y)
			cell.set_highlight_size(CELL_SIZE)
			if (x + y) % 2 == 0:
				cell.sprite.modulate = Color(0.2, 0.2, 0.2, 0.8)
			else:
				cell.sprite.modulate = Color(1, 1, 1, 0.8)
			cell.clicked.connect(_on_cell_clicked)
			row.append(cell)
		cells.append(row)

func _on_cell_clicked(cell, board_pos):
	pass
